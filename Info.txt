if (Target_has_no_debuff:"Rip" and CP:>RipCP and Rip_active) then cast "Rip" //Option "Rip_active": Activate/Deactivate Rip, Option "RipCP": Min. ComboPoints to cast Rip

if (Energy<Reshift_Energylv and RS_active) then cast Reshift // Option "RS_active": Activate or Deactivate Reshift, "Reshift_Energylv": Energy-Level for Reshifting
if (no_my_buff:"Tiger's Fury" and TF_active) then cast "Tiger's Fury" //option: Remaining Time, Option "TF_active": Activate/Deactivate Tiger's Fury
if (Target_is_not_Bleed_Immune and has_no_debuff:"Rake" and (behind and nomybuff:Clearcasting) and Rake_active) cast "Rake" //Option "Rake_active": Activate/Deactivate Rake, (Clearcasting should be used for Shred if Behind)
if ((my_buff:"Clearcasting" or Target_is_Bleed_Immune) and behind and Shred_active) then //Option "Shred_active": Activate/Deactivate Shred
	cast "Shred" 
elseif (Claw_active) // Option "Claw_active": Activate/Deactivate Claw
	cast "Claw"
end if
if (Target_has_no_debuff:"Fearie Fire" and FF_active) then cast "Fearie Fire" //Option "FF_actice": Activate/Deactivate Fearie Fire
